\subsection{Recommended Starting Order}

\begin{alertbox}[\faExclamationTriangle\enspace Action Required — Start Here First]
  \small
  Begin with \textbf{Network Theory} and work through all lectures up to and
  including \textbf{Chapter 3: Transient Analysis}. \\[4pt]
  After that, you may proceed with either \textbf{Digital Electronics} or
  \textbf{Signals \& Systems} — \emph{the choice is yours}.\\[4pt]
  \textbf{Do not start with any other subject initially.}

\subsection{\faDesktop\enspace PC Application Troubleshooting}

\begin{alertbox}[\faExclamationCircle\enspace PC APP Issue? Try These Steps]
  \small
  \begin{enumerate}
    \item \textbf{Disconnect} any external monitor connected to your PC or Mac.
    \item \textbf{Switch} to a different internet connection (hotspot, router, etc.).
    \item \textbf{Re-download} the correct app version:
      \begin{itemize}
        \item Intel chip $\to$ Intel version\quad|\quad Apple chip $\to$ Apple Chip version
      \end{itemize}
    \item Download link is at the \textbf{bottom of our website}.
  \end{enumerate}
\end{alertbox}

\subsection{\faBug\enspace General App Troubleshooting}

\begin{stepbox}[General Fix — Follow in Order]
  \small
  \begin{enumerate}
    \item Mention the exact issue; attach a \textbf{screenshot or short video}
          along with your exact device \& OS (e.g., Windows PC / MacBook M2).
    \item \textbf{Disconnect} any external monitor.
    \item \textbf{Log out from all devices}, then log back in on one device.
    \item Ensure you have the \textbf{latest version} of the app installed.
    \item \textbf{Switch internet connection} (try mobile data or a different Wi-Fi).
    \item Switch browser to \textbf{Google Chrome} if using the website.
  \end{enumerate}
\end{stepbox}
